STEP 1 — DRY RUN (shows orphan episodes)

SELECT
  e.id, e.episode_number, e.title, e.interview_id, e.status,
  (SELECT count(*) FROM tasks t WHERE t.episode_id = e.id) as tasks,
  (SELECT count(*) FROM episode_files f WHERE f.episode_id = e.id) as files,
  (SELECT count(*) FROM episode_shorts s WHERE s.episode_id = e.id) as shorts,
  (SELECT count(*) FROM episode_large_links l WHERE l.episode_id = e.id) as large_links,
  (SELECT count(*) FROM episode_platform_links p WHERE p.episode_id = e.id) as platform_links,
  (SELECT count(*) FROM publishing pub WHERE pub.episode_id = e.id) as publishing_rows
FROM episodes e
WHERE e.interview_id IS NOT NULL
  AND NOT EXISTS (SELECT 1 FROM interviews i WHERE i.id = e.interview_id)
ORDER BY e.episode_number;
